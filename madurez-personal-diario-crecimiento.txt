  <script>
    function initTailwind() {
      tailwind.config = { theme: { extend: { colors: { sage: '#8A9A7B' } } } };
    }
    function startJournal() {
      document.getElementById('journalSection').classList.remove('hidden');
      document.getElementById('journalSection').scrollIntoView({ behavior: 'smooth' });
    }
    function saveJournal() {
      const hp = document.getElementById('highplaceInput').value.trim();
      const rect = document.getElementById('rectitudeInput').value.trim();
      const grow = document.getElementById('growthInput').value.trim();
      localStorage.setItem('maturityJournal2026', JSON.stringify({highplace: hp, rectitude: rect, growth: grow, date: new Date().toISOString()}));
      let html = '';
      if (hp) html += `<li class="flex gap-2"><i class="fas fa-check text-[#8A9A7B]"></i> Lugar alto identificado</li>`;
      if (rect) html += `<li class="flex gap-2"><i class="fas fa-check text-[#8A9A7B]"></i> Rectitud elegida</li>`;
      if (grow) html += `<li class="flex gap-2"><i class="fas fa-check text-[#8A9A7B]"></i> Crecimiento en Cristo</li>`;
      document.getElementById('commitmentsList').innerHTML = html;
      alert('✅ Tu diario de madurez ha sido guardado.');
    }
    function shareExperience() {
      const text = "Hoy escribí en mi 'Diario de Madurez Personal' (2 Reyes 15-16). Estoy quitando mis lugares altos. #DevocionalNataliaZarta";
      navigator.clipboard.writeText(text).then(() => alert('📋 Copiado para redes!'));
    }
    window.onload = initTailwind;
